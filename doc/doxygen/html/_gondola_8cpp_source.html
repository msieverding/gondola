243 float Gondola::calculateCorrectedSpooling(IAnchor *anchor, float targetSpooledDistance)
244 {
245  // Err = b0 * realSpooledRope^2 + b1 * realSpooledRope + b2
246  float b0 = -0.000024216423411;
247  float b1 = 0.051005307386112;
248  float b2 = 0.472932330827063;
249  float ropeOffset = anchor->getRopeOffset();
250  // Error at coordinate zero
251  float estimatedErrZero = b0 * ropeOffset * ropeOffset + b1 * ropeOffset + b2;
252  float ropeSpooledAtTarget = ropeOffset + targetSpooledDistance;
253  // Error at target coordinate
254  float estimatedErrTarget = b0 * ropeSpooledAtTarget * ropeSpooledAtTarget + b1 * ropeSpooledAtTarget + b2;
255  // Estimated Error in movement
256  float estimatedErr = estimatedErrTarget - estimatedErrZero;
257  float correctedDistance = targetSpooledDistance - estimatedErr;
258 
259  logVerbose("Err estimation: estimatedErrZero: %s, estimatedErrTarget %s, estimatedErr %s\n", FTOS(estimatedErrZero), FTOS(estimatedErrTarget), FTOS(estimatedErr));
260  logVerbose("Err estimation: targetSpooledDistance: %s, correctedDistance: %s\n", FTOS(targetSpooledDistance), FTOS(correctedDistance));
261  return correctedDistance;
262 }
